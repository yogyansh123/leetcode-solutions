class Solution:
    def combinationSum(self, candidates, target):
        res = []

        def backtrack(start, path, total):
            if total == target:
                res.append(path[:])
                return

            if total > target:
                return

            for i in range(start, len(candidates)):
                path.append(candidates[i])

                # i pass kiya, kyunki same number unlimited baar use kar sakte hain
                backtrack(i, path, total + candidates[i])

                path.pop()

        backtrack(0, [], 0)
        return res
