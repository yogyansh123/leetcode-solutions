class Solution:
    def canJump(self, nums):
        max_reach = 0

        for i in range(len(nums)):
            # Current index is unreachable
            if i > max_reach:
                return False

            # Update the farthest reachable index
            max_reach = max(max_reach, i + nums[i])

            # Can reach the last index
            if max_reach >= len(nums) - 1:
                return True

        return True
