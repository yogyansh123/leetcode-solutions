class Solution:
    def isNumber(self, s):
        numSeen = False
        dotSeen = False
        eSeen = False

        for i, ch in enumerate(s):

            if ch.isdigit():
                numSeen = True

            elif ch in ['+', '-']:
                if i > 0 and s[i - 1] not in ['e', 'E']:
                    return False

            elif ch == '.':
                if dotSeen or eSeen:
                    return False
                dotSeen = True

            elif ch in ['e', 'E']:
                if eSeen or not numSeen:
                    return False

                eSeen = True
                numSeen = False

            else:
                return False

        return numSeen
