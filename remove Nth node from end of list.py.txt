class Solution:
    def removeNthFromEnd(self, head, n):

        dummy = ListNode(0)
        dummy.next = head

        fast = dummy
        slow = dummy

        # move fast pointer n+1 steps
        for _ in range(n + 1):
            fast = fast.next

        # move both pointers
        while fast:
            fast = fast.next
            slow = slow.next

        # remove nth node
        slow.next = slow.next.next

        return dummy.next
