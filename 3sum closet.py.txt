class Solution:
    def threeSumClosest(self, nums, target):

        nums.sort()

        closest = nums[0] + nums[1] + nums[2]

        for i in range(len(nums) - 2):

            left = i + 1
            right = len(nums) - 1

            while left < right:

                total = nums[i] + nums[left] + nums[right]

                # update closest sum
                if abs(target - total) < abs(target - closest):
                    closest = total

                # exact match
                if total == target:
                    return total

                elif total < target:
                    left += 1

                else:
                    right -= 1

        return closest
