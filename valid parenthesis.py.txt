class Solution:
    def isValid(self, s):

        stack = []

        pairs = {
            ')': '(',
            '}': '{',
            ']': '['
        }

        for ch in s:

            # opening bracket
            if ch in "([{":
                stack.append(ch)

            # closing bracket
            else:
                if not stack or stack[-1] != pairs[ch]:
                    return False

                stack.pop()

        return len(stack) == 0
