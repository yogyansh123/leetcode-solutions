class Solution:
    def jump(self, nums):
        jumps = 0
        current_end = 0
        farthest = 0

        # No need to jump from the last index
        for i in range(len(nums) - 1):
            farthest = max(farthest, i + nums[i])

            # Need to make another jump
            if i == current_end:
                jumps += 1
                current_end = farthest

        return jumps
